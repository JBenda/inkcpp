143private:
144 void BufferDecodedString()
145 {
146 FTCHARToUTF8 Convert(*StringVal);
147 Utf8.SetNum(Convert.Length() + 1);
148 UTF8CHAR* dst = Utf8.GetData();
149 const UTF8CHAR* src = reinterpret_cast<const UTF8CHAR*>(Convert.Get());
150 while (*src) {
151 *dst++ = *src++;
152 }
153 *dst = static_cast<UTF8CHAR>(0);
154 }
